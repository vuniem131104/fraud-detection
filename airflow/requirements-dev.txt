# (postgresql+psycopg) the Feast SQL registry needs (sqlalchemy>=2.0).

# docker-compose.yml, image apache/airflow), so nothing here is needed by the
# main app venv. If you want DAG autocompletion locally, create a dedicated venv:
#
#   python -m venv .venv-airflow
#   .venv-airflow/bin/pip install -r airflow/requirements-dev.txt
#
# Pin these to match the container image (currently apache/airflow:2.10.5).
